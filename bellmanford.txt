INF = float('inf')

def bellman_ford(edges, V, source):
    dist = [INF] * V
    dist[source] = 0

    # Relax edges V-1 times
    for _ in range(V - 1):
        for u, v, w in edges:
            if dist[u] != INF and dist[u] + w < dist[v]:
                dist[v] = dist[u] + w

    # Check for negative weight cycle
    for u, v, w in edges:
        if dist[u] != INF and dist[u] + w < dist[v]:
            print("Negative weight cycle detected!")
            return

    # Print result
    print("Shortest distances from source:")
    for i in range(V):
        print(f"To node {i} = {dist[i]}")


# ------- GRAPH INPUT -------
V = 5  # A, B, C, D, E

edges = [
    (0, 1, -1),  # A -> B
    (0, 2, 4),   # A -> C
    (1, 2, 3),   # B -> C
    (1, 3, 2),   # B -> D
    (1, 4, 2),   # B -> E
    (3, 1, 1),   # D -> B
    (3, 2, 5),   # D -> C
    (4, 3, -3)   # E -> D
]

source = 0  # A

bellman_ford(edges, V, source)
